let selection = {};

function select(product, color) {
  selection[product] = color;
  alert("Selected " + color + " for " + product);
}

function whatsapp(product) {
  let color = selection[product] || "standard";

  let message = "Hello! I want to order: " + product +
                " | Color: " + color;

  let url = "https://wa.me/40712345678?text=" + encodeURIComponent(message);

  window.open(url, "_blank");
}
